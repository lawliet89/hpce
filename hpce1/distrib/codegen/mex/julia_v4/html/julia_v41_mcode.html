1function [viter,vz] = julia_v4(vz, c, maxiter)
 2    
 3    % for code generation poruposes
 4    %vz
 5    assert(isa(vz,'double'));
 6    assert(~isreal(vz));
 7    assert(all(size(vz)<=[1 Inf]));
 8    % c
 9    assert(isa(c,'double'));
10    assert(~isreal(c));
11    assert(isscalar(c));
12    % maxiter 
13    assert(isa(maxiter,'int32'));
14    assert(isscalar(maxiter));
15    
16    viter = ones(1, size(vz, 2));
17    for i = 1:size(vz, 2)
18        while viter(1, i) < maxiter
19            if abs(vz(1, i)) > 1
20                break
21            end;
22            viter(1, i) = viter(1, i) + 1;
23            vz(1, i) = vz(1, i).^2 + c;
24        end
25    end
26end
